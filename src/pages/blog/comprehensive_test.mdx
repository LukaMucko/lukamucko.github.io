## 1. Long Code Block (Real)
Testing vertical scrolling and syntax highlighting for larger scripts.

```python
import random
import time

class DataProcessor:
    """
    A complex class to demonstrate syntax highlighting
    and vertical code block scrolling.
    """
    def __init__(self, seed=42):
        self.seed = seed
        self.data = []
        random.seed(seed)
        
    def generate_noise(self, length=100):
        """Generates random noise data."""
        print(f"Generating {length} points of noise...")
        for i in range(length):
            val = random.gauss(0, 1)
            self.data.append(val)
            
            if i % 10 == 0:
                self._validate_point(val)
                
    def _validate_point(self, val):
        if val > 3:
            return "Outlier"
        elif val < -3:
            return "Outlier"
        return "Normal"
        
    def process(self):
        results = []
        for d in self.data:
            transformed = d ** 2
            results.append(transformed)
        return results

processor = DataProcessor()
processor.generate_noise(50)
print("Processing complete.")
```
